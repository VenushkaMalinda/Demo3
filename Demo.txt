object Demo {
   //Q1
   def celtofar(x:Double):Double=x*1.8+32.0;
   

   //Q2
   def calvol(x:Double):Double=4 * 22 * x * x * x /(3*7);
   

   //Q3
   def calculate(x:Int):Double={
       var n = x*24.95 * 40/100 + 3;
       if(n>50){
           return n + 0.75*(n-50);
       }
       return n;
   }
       
   //main
   def main(args: Array[String]):Unit={
      println(celtofar(35));
      println(calvol(5.0));
      println(calculate(60));
  }
}
